# Minter Checkouts

Minter Checkout is the entry for NFTs. It is used for minting project's NFTs.

### Create a checkout

<Tabs>
<TabItem value="js" label="JavaScript">

```js
import { InteractWith } from '@interactwith/node';

const interactwith = new InteractWith({
  apiKey: 'YOUR_API_KEY',
})

const checkout = await interactwith.minter.minter_checkouts.create({
  name: 'Checkout 1',
  description: 'Checkout 1 description',
  price: 100,
  currency: 'USD',
  project: 'PROJECT_ID',
  metadata: {
    name: 'Checkout 1',
    description: 'Checkout 1 description',
    image: 'https://example.com/image.png',
  },
})
```
</TabItem>
<TabItem value="gql" label="GraphQL">

```graphql

mutation {
  createMinterCheckout(
    input: {
      name: "Checkout 1",
      description: "Checkout 1 description",
      price: 100,
      currency: "USD",
      project: "PROJECT_ID",
      metadata: {
        name: "Checkout 1",
        description: "Checkout 1 description",
        image: "https://example.com/image.png",
      },
    }
  ) {
    id
    name
    description
    price
    currency
    project
    metadata
    created_at
    updated_at
  }
}

```

</TabItem>

</Tabs>

### Get a checkout

<Tabs>
<TabItem value="js" label="JavaScript">

```js
import { InteractWith } from '@interactwith/node';

const interactwith = new InteractWith({
  apiKey: "API_KEY",
});

const checkout = await interactwith.minter.minter_checkouts.get({
  id: "CHECKOUT_ID",
});
```

</TabItem>
<TabItem value="gql" label="GraphQL">

```graphql
query {
  minter_checkout(id: "CHECKOUT_ID") {
    id
    name
    description
    price
    currency
    project
    metadata
    created_at
    updated_at
  }
}
```

</TabItem>

</Tabs>


### List checkouts

<Tabs>
<TabItem value="js" label="JavaScript">

```js
import { InteractWith } from '@interactwith/node';

const interactwith = new InteractWith({
  apiKey: "API_KEY",
});

const checkouts = await interactwith.minter.minter_checkouts.list({
  project: "PROJECT_ID",
});
```

</TabItem>
<TabItem value="gql" label="GraphQL">

```graphql
query {
  minter_checkouts(
    filter: {
      project: "PROJECT_ID"
    },
    pagination: {
      page: 1,
      take: 30,
    },
    sort: {
      by_field: "created_at",
      direction: "DESC",
    },
  ) {
    id
    name
    description
    price
    currency
    project
    metadata
    created_at
    updated_at
  }
}
```

</TabItem>
